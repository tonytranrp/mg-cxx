+void Sema::BeginCXXMGConditionalMemberArm(CXXRecordDecl *TargetRecord,
+                                          AccessSpecifier AS) {
+  assert(TargetRecord && "conditional member arm needs a target record");
+
+  CXXMGConditionalMemberArmState State;
+  State.TargetRecord = TargetRecord;
+  State.StartAS = AS;
+  CXXMGConditionalMemberArmStack.push_back(std::move(State));
+}
+
+void Sema::EndCXXMGConditionalMemberArm(
+    llvm::SmallVectorImpl<Decl *> &OutPatternDecls) {
+  assert(!CXXMGConditionalMemberArmStack.empty() &&
+         "ending conditional member arm without active state");
+
+  CXXMGConditionalMemberArmState &State =
+      CXXMGConditionalMemberArmStack.back();
+  OutPatternDecls.append(State.PatternDecls.begin(), State.PatternDecls.end());
+  CXXMGConditionalMemberArmStack.pop_back();
+}
+
+bool Sema::isCollectingCXXMGConditionalMemberForCurrentContext() const {
+  return !CXXMGConditionalMemberArmStack.empty() &&
+         CXXMGConditionalMemberArmStack.back().TargetRecord == CurContext;
+}
+
+void Sema::AddCXXMGConditionalMemberPatternDecl(Decl *D) {
+  assert(D && "cannot collect a null conditional member pattern declaration");
+  assert(isCollectingCXXMGConditionalMemberForCurrentContext() &&
+         "collecting a conditional member pattern from the wrong context");
+  CXXMGConditionalMemberArmStack.back().PatternDecls.push_back(D);
+}
+
+Decl *Sema::ActOnCXXMGConditionalMember(
+    Scope *S, Decl *TagDecl, SourceLocation IfLoc,
+    llvm::ArrayRef<CXXMGConditionalMemberArm> Arms) {
+  CXXRecordDecl *Record = dyn_cast_or_null<CXXRecordDecl>(TagDecl);
+  if (!Record)
+    if (auto *Template = dyn_cast_or_null<ClassTemplateDecl>(TagDecl))
+      Record = Template->getTemplatedDecl();
+  if (!Record)
+    return nullptr;
+
+  auto *D = CXXMGConditionalMemberDecl::Create(Context, Record, IfLoc, Arms);
+  Record->addDecl(D);
+
+  for (const CXXMGConditionalMemberArm &Arm : D->arms()) {
+    bool Selected = Arm.isElseArm();
+    if (Arm.hasCondition()) {
+      if (Arm.Condition->isValueDependent())
+        break;
+
+      bool ConditionValue = false;
+      if (!Arm.Condition->EvaluateAsBooleanCondition(ConditionValue, Context))
+        continue;
+
+      Selected = ConditionValue;
+    }
+
+    if (!Selected)
+      continue;
+
+    for (Decl *Pattern : Arm.Decls) {
+      Record->addDecl(Pattern);
+      if (auto *BUD = dyn_cast<BaseUsingDecl>(Pattern))
+        for (UsingShadowDecl *Shadow : BUD->shadows())
+          Record->addDecl(Shadow);
+      if (auto *FD = dyn_cast<FieldDecl>(Pattern))
+        FieldCollector->Add(FD);
+    }
+    break;
+  }
+
+  return D;
+}
